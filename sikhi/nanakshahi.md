ਨਾਨਕਸ਼ਾਹੀ ਕੈਲੰਡਰ · Gurpurabs, itihasik dihade, sangrands, massia and purnmashi, following the official Nanakshahi Jantri (Amritsar). Years the Jantri has published are **confirmed** (extracted page-by-page from the printed calendar); future years are **≈ estimated** from a calibrated astronomical model until the real Jantri appears. This data is generated by [nanakshahi-ical](https://github.com/janpreet/nanakshahi-ical), a separate open-source project; this page just renders it in the site's own look.

<p class="ns-nt-note">A <b>pahar</b> (ਪਹਿਰ) is a three-hour watch, eight in a full day, four in the night,
which runs sunset to sunrise. <b>Amritvela</b> is the last pahar of the night: the Sikh Rehat Maryada
(Art. 4) places it about three hours before dawn. <b>Rehraas Sahib</b> is read at sunset and
<b>Kirtan Sohila</b> before sleeping. <span id="ns-ntDetail">Times shown are the fixed feed times.</span>
The subscribable <code>nitnem.ics</code> always uses the fixed global times 03:30 · 18:30 · 21:00.</p>

<script>
(function() {
  const DATA_BASE = 'https://janpreet.github.io/nanakshahi-ical/';
  const MONTH_PA = { Chet:'ਚੇਤ', Vaisakh:'ਵੈਸਾਖ', Jeth:'ਜੇਠ', Harh:'ਹਾੜ', Sawan:'ਸਾਵਣ', Bhadon:'ਭਾਦੋਂ', Assu:'ਅੱਸੂ', Katak:'ਕੱਤਕ', Maghar:'ਮੱਘਰ', Poh:'ਪੋਹ', Magh:'ਮਾਘ', Phagan:'ਫੱਗਣ' };
  const fmt = iso => new Date(iso + 'T00:00:00Z').toLocaleDateString('en-GB', { day:'numeric', month:'short', year:'numeric', timeZone:'UTC' });
  const esc = s => (s ?? '').replace(/[&<>"]/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c]));

  fetch(DATA_BASE + 'data.json').then(r => r.json()).then(data => {
    const years = Object.keys(data.years).sort();
    const tabs = document.getElementById('ns-tabs');
    const nowIso = new Date().toISOString().slice(0, 10);
    let current = years.find(y => data.years[y].months[0].start <= nowIso && nowIso < data.years[y].nextChet) ?? years[0];
    for (const y of years) {
      const b = document.createElement('button');
      b.role = 'tab'; b.textContent = 'NS ' + y + (data.years[y].status === 'estimated' ? ' ≈' : '');
      b.onclick = () => { current = y; render(); };
      b.dataset.year = y;
      tabs.appendChild(b);
    }
    function render() {
      for (const b of tabs.children) b.setAttribute('aria-selected', b.dataset.year === current);
      const yr = data.years[current];
      const view = document.getElementById('ns-view');
      const est = yr.status === 'estimated';
      let html = `<span class="ns-banner ${est ? 'est' : 'ok'}">${est ? '≈ ESTIMATED — computed, awaiting the published Jantri' : 'CONFIRMED — extracted from the published Jantri'}</span>`;
      html += `<p class="ns-src">${est
        ? 'Dates for this year are computed from the calibrated astronomical model. They will be confirmed against the printed Jantri once it is published, until then they carry the ≈ mark.'
        : 'Every date below was read from the official printed Jantri for this year; hover a ✓ for its page reference.'}</p>`;
      html += `<p class="ns-legend">Bold = Gurpurab. Each month card shows its sangrand and length; ਮ = massia, ਪ = purnmashi.</p>`;
      html += '<div class="ns-months">';
      for (const m of yr.months) {
        const end = new Date(Date.parse(m.start) + (m.days - 1) * 86400000).toISOString().slice(0, 10);
        const evs = yr.events.filter(e => e.date >= m.start && e.date <= end);
        const mas = yr.lunar.massia.filter(d => d >= m.start && d <= end).map(fmt);
        const pur = yr.lunar.purnmashi.filter(d => d >= m.start && d <= end).map(fmt);
        html += `<div class="ns-month"><h3>${MONTH_PA[m.name]} ${m.name} <span class="ns-meta">· ${current}</span></h3>
          <div class="ns-meta">1 ${m.name} = ${fmt(m.start)} · ${m.days} days<br>ਮ ${mas.join(', ') || '—'} · ਪ ${pur.join(', ') || '—'}</div>`;
        const byDate = new Map();
        for (const e of evs) {
          if (!byDate.has(e.date)) byDate.set(e.date, []);
          byDate.get(e.date).push(e);
        }
        for (const [date, group] of byDate) {
          html += `<div class="ns-ev"><span class="ns-d">${fmt(date)}<br>${esc(group[0].ns ?? '')}</span><span class="ns-stack">`;
          for (const e of group) {
            html += `<span class="ns-ev-item ${e.category === 'gurpurab' ? 'g' : ''}">
              <span>${esc(e.en)}${e.pa ? '<br><span style="color:var(--text-muted)">' + esc(e.pa) + '</span>' : ''}</span>
              <span class="ns-chip ${e.status === 'estimated' ? 'est' : 'ok'}" title="${esc(e.detail ?? '')}">${e.status === 'estimated' ? '≈' : '✓'}</span>
            </span>`;
          }
          html += '</span></div>';
        }
        html += '</div>';
      }
      html += '</div>';
      view.innerHTML = html;
    }
    document.getElementById('ns-icsUrl').textContent = DATA_BASE + 'nanakshahi.ics';
    document.getElementById('ns-nitnemUrl').textContent = DATA_BASE + 'nitnem.ics';
    render();
  });

  // Localized nitnem times: resolve the visitor's IANA timezone to its tzdb
  // reference-city coordinates (no geolocation prompt) and compute today's
  // sunrise/sunset there (NOAA algorithm). Amritvela = sunrise − 3h (Rehat
  // Maryada), Rehraas = sunset. Falls back silently to the fixed feed times.
  fetch(DATA_BASE + 'tzcoords.json').then(r => r.json()).then(coords => {
    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;
    const [lat, lon] = coords[tz] ?? [];
    if (lat === undefined) return;
    const rad = d => d * Math.PI / 180, deg = r => r * 180 / Math.PI;
    function sunUTC(rising) {
      const now = new Date();
      const [y, m, d] = [now.getFullYear(), now.getMonth(), now.getDate()];
      const N = Math.floor((Date.UTC(y, m, d) - Date.UTC(y, 0, 0)) / 86400000);
      const lngHour = lon / 15;
      const t = N + ((rising ? 6 : 18) - lngHour) / 24;
      const M = 0.9856 * t - 3.289;
      let L = M + 1.916 * Math.sin(rad(M)) + 0.020 * Math.sin(rad(2 * M)) + 282.634;
      L = ((L % 360) + 360) % 360;
      let RA = ((deg(Math.atan(0.91764 * Math.tan(rad(L)))) % 360) + 360) % 360;
      RA = (RA + (Math.floor(L / 90) - Math.floor(RA / 90)) * 90) / 15;
      const sinDec = 0.39782 * Math.sin(rad(L));
      const cosH = (Math.cos(rad(90.833)) - sinDec * Math.sin(rad(lat))) /
        (Math.cos(Math.asin(sinDec)) * Math.cos(rad(lat)));
      if (cosH > 1 || cosH < -1) return null; // polar day/night
      const H = (rising ? 360 - deg(Math.acos(cosH)) : deg(Math.acos(cosH))) / 15;
      const UT = (((H + RA - 0.06571 * t - 6.622 - lngHour) % 24) + 24) % 24;
      return new Date(Date.UTC(y, m, d) + UT * 3600000);
    }
    const sunrise = sunUTC(true), sunset = sunUTC(false);
    if (!sunrise || !sunset) return;
    const hm = t => t.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', hour12: false });
    document.getElementById('ns-ntAmrit').textContent = '≈ ' + hm(new Date(sunrise - 3 * 3600000));
    document.getElementById('ns-ntAmritWhy').textContent = `3 h before sunrise (${hm(sunrise)})`;
    document.getElementById('ns-ntRehraas').textContent = '≈ ' + hm(sunset);
    document.getElementById('ns-ntRehraasWhy').textContent = 'at sunset';
    document.getElementById('ns-ntPlace').textContent = '— ' + tz.split('/').pop().replace(/_/g, ' ');
    document.getElementById('ns-ntDetail').textContent =
      `Times above are computed for today in your timezone (${tz}, from its reference city, hence ≈).`;
  }).catch(() => {});
})();
</script>
